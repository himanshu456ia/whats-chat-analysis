// Helper function for DFS to check for odd cycles
bool isCyclicDFS(struct LinkedList* adjacencyList, int vertex, int parent, int* visited, int* level) {
    visited[vertex] = 1; // Mark the current vertex as visited
    for (struct Node* current = adjacencyList[vertex].head; current; current = current->next) {
        int neighbor = current->data;
        if (!visited[neighbor]) {
            level[neighbor] = level[vertex] + 1;
            if (isCyclicDFS(adjacencyList, neighbor, vertex, visited, level))
                return true;
        } else if (neighbor != parent && (level[vertex] - level[neighbor]) % 2 == 0) {
            // If the neighbor is not the parent and the level difference is even, it's an odd cycle
            return true;
        }
    }
    return false;
}

// Function to check if the graph is bipartite (contains no odd cycles)
bool isBipartite(struct LinkedList* adjacencyList, int n) {
    int* visited = (int*)malloc(n * sizeof(int));
    int* level = (int*)malloc(n * sizeof(int));

}

// Helper function for DFS to find articulation points (critical vertices)
void findArticulationPoints(struct LinkedList* adjacencyList, int vertex, int* visited, int* discTime, int* lowTime, int* parent, bool* isArticulation) {
    static int time = 0;
    int children = 0;

    visited[vertex] = 1;
    discTime[vertex] = lowTime[vertex] = ++time;

    for (struct Node* current = adjacencyList[vertex].head; current; current = current->next) {
        int neighbor = current->data;

        if (!visited[neighbor]) {
            children++;
            parent[neighbor] = vertex;

            findArticulationPoints(adjacencyList, neighbor, visited, discTime, lowTime, parent, isArticulation);

            lowTime[vertex] = (lowTime[vertex] < lowTime[neighbor]) ? lowTime[vertex] : lowTime[neighbor];

            if (parent[vertex] == -1 && children > 1) {
                isArticulation[vertex] = true;
            }

            if (parent[vertex] != -1 && lowTime[neighbor] >= discTime[vertex]) {
                isArticulation[vertex] = true;
            }
        } else if (neighbor != parent[vertex]) {
            lowTime[vertex] = (lowTime[vertex] < discTime[neighbor]) ? lowTime[vertex] : discTime[neighbor];
        }
    }
}

using namespace std;

int solve(int N, vector<int>& A, vector<int>& B) {
    vector<vector<int>> dp(N + 1, vector<int>(N + 1, 0));

    
    for (int i = 1; i <= N; ++i) {
        for (int j = 1; j <= N; ++j) {
            if (A[i - 1] == B[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
            }
        }
    }

 
    int L = dp[N][N];

    
    int turns = 2 * (N - L);

    return turns;
}
